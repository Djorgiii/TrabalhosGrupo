Análise de Biblioteca de Áudio
Dashboard analítico para coleção de ficheiros de áudio
audioData = FileAttachment("data/audio-data.json").json()

// Filtros interativos
viewof generoSelecionado = Inputs.select(
  ["Todos", ...new Set(audioData.map(d => d.genero))],
  {label: "Filtrar por Género:", value: "Todos"}
)

viewof artistaSelecionado = Inputs.select(
  ["Todos", ...new Set(audioData.map(d => d.artista))],
  {label: "Filtrar por Artista:", value: "Todos"}
)

// Dados filtrados
dadosFiltrados = audioData.filter(d => 
  (generoSelecionado === "Todos" || d.genero === generoSelecionado) &&
  (artistaSelecionado === "Todos" || d.artista === artistaSelecionado)
)

// Métricas calculadas
totalMusicas = dadosFiltrados.length
duracaoTotal = d3.sum(dadosFiltrados, d => d.duracao)
duracaoMedia = d3.mean(dadosFiltrados, d => d.duracao)
popularidadeMedia = d3.mean(dadosFiltrados, d => d.popularidade)

// Importar D3 e Plot
d3 = require("d3@7")
Plot = require("@observablehq/plot")
