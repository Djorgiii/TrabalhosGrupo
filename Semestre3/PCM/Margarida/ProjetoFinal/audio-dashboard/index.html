viewof selection = {
  const container = html`<div class="playlist-container"></div>`;
  
  container.value = data[0];

  for (const song of data) {
    const row = html`
      <div style="
        display: flex; 
        align-items: center; 
        padding: 10px; 
        background: #0f2747; 
        border-radius: 8px; 
        cursor: pointer; 
        color: #eef4ff; 
        transition: background 0.2s, transform 0.1s;">
        
        <img src="${song.img}" style="width: 50px; height: 50px; border-radius: 4px; margin-right: 15px; object-fit: cover;">
        
        <div>
          <div style="font-weight: bold; font-size: 1.1em;">${song.Música}</div>
          <div style="opacity: 0.7; font-size: 0.9em;">${song.Artista}</div>
        </div>
      </div>
    `;

    row.onclick = () => {
      for (const child of container.children) child.style.background = "#0f2747";
      row.style.background = "#1a3664"; // Azul selecionado
      container.value = song;
      container.dispatchEvent(new Event("input", {bubbles: true}));
    };

    container.append(row);
  }

  if (container.children[0]) container.children[0].style.background = "#1a3664";

  // Garante que o valor inicial é emitido, evitando undefined no player
  container.dispatchEvent(new Event("input", {bubbles: true}));

  return container;
}

millisToMinutesAndSeconds = (millis) => {
  if (!millis) return "-";
  const minutes = Math.floor(millis / 60000);
  const seconds = ((millis % 60000) / 1000).toFixed(0);
  return minutes + ":" + (seconds < 10 ? '0' : '') + seconds;
}

createTag = (label, value, icon = "") => {
  return html`

      <div style="font-size: 1rem; font-weight: 600; color: var(--text-primary);">
        ${icon} ${value || "-"}
      </div>
    </div>
  `
}
